\section{语义区块环境}

% --- 定理与定义 ---
\begin{frame}{定理与定义 --- \texttt{thm} / \texttt{defn}}
  \begin{thm}[动能定理]
  	合外力对物体做的功等于物体动能的变化量：
  	\[ W_{\text{合}} = E_{k2} - E_{k1} = \frac{1}{2}mv_2^2 - \frac{1}{2}mv_1^2 \]
  \end{thm}
	\vfil
  \begin{defn}[加速度]
    加速度是描述物体速度变化快慢的物理量，等于速度的变化量与发生这一变化所用时间的比值：
    \[ \vec{a} = \frac{\Delta \vec{v}}{\Delta t} \]
  \end{defn}

  
\end{frame}

% --- 重点与警示 ---
\begin{frame}{重点与警示 --- \texttt{imp} / \texttt{warn}}
  \begin{imp}[核心考点]
    动量定理是矢量式，必须规定正方向！解题步骤：
    \begin{enumerate}
      \item 选取研究对象
      \item 规定正方向
      \item 分析初末状态的动量
      \item 代入公式 $\vec{I} = \Delta \vec{p}$
    \end{enumerate}
  \end{imp}
	\vfil
  \begin{warn}[易错警示]
    计算冲量时，\hlwarn{绝不能}直接用 $I = Ft$，除非 $F$ 是恒力！
    对于变力，只能用动量定理求平均力。
  \end{warn}

\end{frame}

    质量为 $m$ 的物体在倾角为 $\theta$ 的斜面上匀速下滑，求动摩擦因数 $\mu$。
  \end{eg}
\vfil

  \end{ex}

% --- 技巧与拓展 ---
\begin{frame}{技巧与拓展 --- \texttt{tip} / \texttt{supp}}
  \begin{tip}[解题技巧]
    运用牛顿第二定律时，\hl{正交分解}是最核心的方法：
    \begin{itemize}
      \item 沿运动方向建立 $x$ 轴
      \item 垂直运动方向建立 $y$ 轴
      \item 分别列出分量方程
    \end{itemize}
  \end{tip}
\vfil
  \begin{supp}[拓展知识]
    如果斜面不是固定的，而是放在光滑水平面上，则需要考虑系统质心运动定理\footnote{此时两者构成的系统在水平方向动量守恒。}。
  \end{supp}
\end{frame}

% --- 备课笔记与解答 ---
\begin{frame}{备课笔记与解答 --- \texttt{memo} / \texttt{solve}}
  \begin{memo}[备课笔记]
    学生常犯错误：忘记 $\Delta \vec{v}$ 是末速度减初速度，尤其是反弹问题中
    速度方向改变导致的符号问题。建议用具体数值示例反复强调。
  \end{memo}

  \begin{solve}
    以小球为研究对象，规定向下为正方向。

    落地前速度：$v_1 = \sqrt{2gh_1} = \sqrt{2 \times 10 \times 1.8} = 6\Ums$

    反弹后速度：$v_2 = -\sqrt{2gh_2} = -\sqrt{2 \times 10 \times 0.8} = -4\Ums$

    由动量定理：$\bar{F}\Delta t = m v_2 - m v_1$

    $\vec{F} = \dfrac{0.5 \times (-4 - 6)}{0.01} \UN= -500\UN$
  \end{solve}
\end{frame}

% --- 区块 + 叠加动画 ---
\begin{frame}{区块在叠加动画中的表现}
  \begin{thm}[牛顿第二定律]
    \[ \vec{F} = m \vec{a} \]
  \end{thm}
\vfil
  \pause

  \begin{imp}
    加速度 $\vec{a}$ 的方向与合外力 $\vec{F}_{\text{合}}$ 的方向相同，
    \hl{而非}速度 $\vec{v}$ 的方向。
  \end{imp}
\vfil
  \pause

  \begin{warn}
    公式中的 $m$ 是物体的\hlwarn{惯性质量}，与引力质量等价。
  \end{warn}
\end{frame}

% --- 练习 + 解答 ---
\begin{frame}{练习 + 解答展示}
  \begin{ExerciseList}

    \item 两个质量分别为 $m_1$、$m_2$ 的物体在光滑水平面上发生完全非弹性碰撞，
          已知碰前速度分别为 $v_1$、$v_2$（同方向），求碰后共同速度和动能损失。

    \jdanswer{
      由动量守恒：$m_1 v_1 + m_2 v_2 = (m_1 + m_2)v$\\
      碰后共同速度：
      $$
      	v = \frac{m_1 v_1 + m_2 v_2}{m_1 + m_2}
      $$
      动能损失：$$\Delta E_k = \frac{1}{2}\dfrac{m_1 m_2}{m_1 + m_2}(v_1 - v_2)^2$$
    }
  \end{ExerciseList}
\end{frame}

\begin{frame}{填空题 --- \texttt{\textbackslash tkanswer}}
  牛顿第二定律的数学表达式为 \tkanswer{$\vec{F}=m\vec{a}$}。

  \vspace{1em}

  动能的定义式为 \tkanswer{$E_k = \dfrac{1}{2}mv^2$}，国际单位是 \tkanswer{焦耳，符号$\UJ[a]$}。

  \vspace{1em}

  万有引力定律的表达式为 \tkanswer{$F = G\dfrac{m_1 m_2}{r^2}$}，其中引力常量
  $G$ 的数值为 \tkanswer{$6.67 \times 10^{-11}\UNmqkgq$}。

  \vspace{1em}

  电荷在电场中的电势能变化量 $\Delta E_p =$ \tkanswer{$-W_{\text{电场力}}$}，
  电场力做功与路径 \tkanswer{无关}。
\end{frame}

% --- 脚注 ---
\begin{frame}{脚注与参考文献}
  牛顿第二定律\footnote{这是动量定理在质量不变时的特例。}是经典力学的核心定律之一。

  万有引力定律\footnote{适用于任意两个质点之间，引力方向沿两质点连线。}揭示了天体运行与地面物体运动的统一规律。

  质点\footnote{在某些问题中，可忽略物体的形状和大小，视为有质量的点。}是力学中常用的简化模型。

  \begin{itemize}
    \item 气垫导轨\footnote{利用压缩空气在导轨与滑块之间形成气膜，大幅减小摩擦力的实验装置。}是验证动量定理的常用设备。
    \item 光电门\footnote{通过遮光条挡住光线的时间来测量物体的速度。遮光条越窄，测量的瞬时速度越精确。}用于测量瞬时速度。
  \end{itemize}
\end{frame}

\section{综合示例：动量守恒}

\begin{frame}{动量守恒定律}
  \begin{thm}[动量守恒定律]
    一个系统不受外力或所受外力的矢量和为零时，这个系统的总动量保持不变：
    \[ m_1 \vec{v}_1 + m_2 \vec{v}_2 = m_1 \vec{v}_1' + m_2 \vec{v}_2' \]
  \end{thm}

  \begin{imp}[适用条件]
    \begin{enumerate}
      \item 系统不受外力或合外力为零
      \item 内力远大于外力（如碰撞、爆炸）
      \item 某方向合外力为零，则该方向动量守恒
    \end{enumerate}
  \end{imp}
\end{frame}

\begin{frame}{动量守恒 --- 典例分析}
  \begin{eg}[人船模型]
    质量为 $M$ 的船静止在水面上，船上站着一个质量为 $m$ 的人，
    当人相对于地面以速度 $v$ 向右走时，求船的运动情况。
  \end{eg}

  \begin{solve}
    选人、船为系统。水平方向不受外力，动量守恒。

    设船的速度为 $V$（方向向左）。

    由动量守恒：$mv + M(-V) = 0$

    $\Rightarrow V = \dfrac{m}{M}v$（方向与人的速度方向相反）
  \end{solve}
\end{frame}
