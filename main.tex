\section{语义区块环境}

\begin{frame}{定理与定义 --- \texttt{thm} / \texttt{defn}}
  \begin{thm}[勾股定理]
    直角三角形两直角边的平方和等于斜边的平方：
    \[ a^2 + b^2 = c^2 \]
  \end{thm}

  \begin{defn}[加速度]
    加速度是描述物体速度变化快慢的物理量，等于速度的变化量与发生这一变化所用时间的比值：
    \[ \vec{a} = \frac{\Delta \vec{v}}{\Delta t} \]
  \end{defn}
\end{frame}

\begin{frame}{重点与警示 --- \texttt{imp} / \texttt{warn}}
  \begin{imp}[核心考点]
    动量定理是矢量式，必须规定正方向！解题步骤：
    \begin{enumerate}
      \item 选取研究对象
      \item 规定正方向
      \item 分析初末状态的动量
      \item 代入公式 $\vec{I} = \Delta \vec{p}$
    \end{enumerate}
  \end{imp}

  \begin{warn}[易错警示]
    计算冲量时，\hlwarn{绝不能}直接用 $I = Ft$，除非 $F$ 是恒力！
    对于变力，只能用动量定理求平均力。
  \end{warn}
\end{frame}

\begin{frame}{例题与练习 --- \texttt{eg} / \texttt{ex}}
  \begin{eg}[斜面问题]
    质量为 $m$ 的物体在倾角为 $\theta$ 的斜面上匀速下滑，求动摩擦因数 $\mu$。
  \end{eg}

  \begin{ex}[变力问题]
    质量为 $0.5\Ukg$ 的小球从 $1.8\Um$ 高处自由落体，碰地后反弹高度为 $0.8\Um$，
    碰撞时间为 $0.01\Us$，求碰撞过程中的平均作用力。
  \end{ex}
\end{frame}

\begin{frame}{技巧与拓展 --- \texttt{tip} / \texttt{supp}}
  \begin{tip}[解题技巧]
    运用牛顿第二定律时，\hl{正交分解}是最核心的方法：
    \begin{itemize}
      \item 沿运动方向建立 $x$ 轴
      \item 垂直运动方向建立 $y$ 轴
      \item 分别列出分量方程
    \end{itemize}
  \end{tip}

  \begin{supp}[拓展知识]
    如果斜面不是固定的，而是放在光滑水平面上，则需要考虑系统质心运动定理\footnote{此时两者构成的系统在水平方向动量守恒。}。
  \end{supp}
\end{frame}

\begin{frame}{备课笔记与解答 --- \texttt{memo} / \texttt{solve}}
  \begin{memo}[备课笔记]
    学生常犯错误：忘记 $\Delta \vec{v}$ 是末速度减初速度，尤其是反弹问题中
    速度方向改变导致的符号问题。建议用具体数值示例反复强调。
  \end{memo}

  \begin{solve}
    以小球为研究对象，规定向下为正方向。

    落地前速度：$v_1 = \sqrt{2gh_1} = \sqrt{2 \times 10 \times 1.8} = 6\Ums$

    反弹后速度：$v_2 = -\sqrt{2gh_2} = -\sqrt{2 \times 10 \times 0.8} = -4\Ums$

    由动量定理：$\bar{F}\Delta t = m v_2 - m v_1$

    $\bar{F} = \frac{0.5 \times (-4 - 6)}{0.01} = -500\UN$
  \end{solve}
\end{frame}

\begin{frame}{练习 + 解答展示}
  \begin{ExerciseList}
    \item 质量为 $m$ 的小球从高度 $h$ 处自由下落，与地面碰撞后反弹至 $\frac{h}{4}$ 处，
          碰撞时间为 $t$，求地面对小球的平均作用力。

    \jdanswer{
      规定向下为正方向。\\
      落地前速度：$v_1 = \sqrt{2gh}$\\
      反弹后速度：$v_2 = -\sqrt{2g \cdot h/4} = -\frac{1}{2}\sqrt{2gh}$\\
      由动量定理：$(mg - \bar{N})t = m(v_2 - v_1)$\\
      $\bar{N} = mg + \frac{3m\sqrt{2gh}}{2t}$
    }

    \item 两个质量分别为 $m_1$、$m_2$ 的物体在光滑水平面上发生完全非弹性碰撞，
          已知碰前速度分别为 $v_1$、$v_2$（同方向），求碰后共同速度和动能损失。

    \jdanswer{
      由动量守恒：$m_1 v_1 + m_2 v_2 = (m_1 + m_2)v$\\
      碰后共同速度：$v = \frac{m_1 v_1 + m_2 v_2}{m_1 + m_2}$\\
      动能损失：$\Delta E_k = \frac{1}{2}\frac{m_1 m_2}{m_1 + m_2}(v_1 - v_2)^2$
    }
  \end{ExerciseList}
\end{frame}

    {曲线运动所受合外力一定不为零}

\begin{frame}{填空题 --- \texttt{\textbackslash tkanswer}}
  牛顿第二定律的数学表达式为 \tkanswer{$\vec{F}=m\vec{a}$}。

  \vspace{1em}

  \vspace{1em}

\end{frame}

\begin{frame}{脚注与参考文献}
  牛顿第二定律\footnote{这是动量定理在质量不变时的特例。}是经典力学的核心定律之一。

  万有引力定律\footnote{适用于任意两个质点之间，引力方向沿两质点连线。}揭示了天体运行与地面物体运动的统一规律。

  质点\footnote{在某些问题中，可忽略物体的形状和大小，视为有质量的点。}是力学中常用的简化模型。

  \begin{itemize}
    \item 气垫导轨\footnote{利用压缩空气在导轨与滑块之间形成气膜，大幅减小摩擦力的实验装置。}是验证动量定理的常用设备。
    \item 光电门\footnote{通过遮光条挡住光线的时间来测量物体的速度。遮光条越窄，测量的瞬时速度越精确。}用于测量瞬时速度。
  \end{itemize}
\end{frame}

\begin{frame}{双栏布局 --- \texttt{columns} 环境}
  \begin{columns}[T]
    \begin{column}{0.48\linewidth}
      \begin{defn}[动能]
        物体因运动而具有的能量：
        \begin{equation}
          E_k = \frac{1}{2}mv^2
        \end{equation}
      \end{defn}

      \begin{tip}
        动能是标量（无方向性），单位与功相同，为焦耳（$\UJ$）。
      \end{tip}
    \end{column}

    \begin{column}{0.48\linewidth}
      \begin{defn}[重力势能]
        物体被举高而具有的能量（选地面为零势面）：
        \begin{equation}
          E_p = mgh
        \end{equation}
      \end{defn}

      \begin{warn}
        势能是相对量！零势面的选取\hl{不同}，数值不同。
      \end{warn}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{\texttt{description} 环境}
  使用 \texttt{description} 环境对术语进行解释：

  \begin{description}
    \item[质点] 在某些问题中，可忽略物体的形状和大小，视为有质量的点。
    \item[刚体] 在任何外力作用下，形状和大小保持不变的物体（理想模型）。
    \item[惯性系] 牛顿运动定律成立的参考系。
    \item[非惯性系] 牛顿运动定律不成立的参考系，需引入惯性力。
  \end{description}
\end{frame}
